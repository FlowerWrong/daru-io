def load(filename, fields)
  ds = Daru::DataFrame.new({}, order: fields)
  File.open(filename,'r').each_line do |line|
    row = PlaintextHelper.process_row(line.strip.split(/\s+/),[''])
    next if row == ["\x1A"]
    ds.add_row(row)
  end
  ds.update
  fields.each { |f| ds[f].rename f }
  ds
end
